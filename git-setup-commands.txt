# After creating GitHub repository, run these commands:

# Add the GitHub repository as origin (replace YOUR_USERNAME)
git remote add origin https://github.com/YOUR_USERNAME/havila-maritime-platform.git

# Push your code to GitHub
git push -u origin main

# When prompted for credentials:
# Username: your-github-username
# Password: paste-your-personal-access-token-here

# Alternative: Store credentials (optional)
git config credential.helper store
